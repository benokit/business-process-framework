------------ messaging transactional outbox ---------------

implementation/infrastructure/messaging/

elements:

{
    type: service,
    id: transactional-outbox-service,
    interface: {
        put: {
            !destination: string,
            !envelope: @message-envelope 
        }
    }
}

transactional outbox item:
    - destination
    - retryCount
    - processAfterTimestampUTC
    - status: 
        - 0: waiting
        - 1: processed
        - 2: failed
    - envelope
        - timestampUTC
        - group
        - message
        - ...

initial transactional outbox item:
    - destination = input.destination
    - retryCount = 0
    - status = 0
    - processAfterTimestampUTC = input.envelope.timestampUTC 
    - envelope = input.envelope

items are stored in mongodb collection: transactional-outbox

there is transactional outbox processor service 
- methods
    - run 
        - lockIntervalInMilliseconds
        - idleIntervalInMilliseconds
    - stop

transactional outbox processor loop:

- inTransaction: {
 - read item from transactional outbox: 
     let firstInGroups = map (head . (sortBy message.timestampUTC asc)) $ (groupBy message.group $ filter status = 0 $ items)
     item =  head . (sortBy processAfterTimestampUTC desc) $ firstInGroups
     return item 
 - update item.processAfterTimestampUTC += lockIntervalInMilliseconds
}
- try 
  - publish item.envelope on item.destination
  - update:
    - item.status = 1
- catch 
  - if doRetry(destination, item.retryCount)
    - update
      - item.retryCount += 1
      - item.processAfterTimestampUTC += computeRetryDelay(item.retryCount, destination.publish.retry)
  - else 
    - update 
      - item.status = 2
